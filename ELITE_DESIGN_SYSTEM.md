```tsx
        {period && (
          <Text style={styles.changePeriod}>vs {period}</Text>
        )}
      </View>
    )}
    
    {chart && chartData && (
      <View style={styles.chartContainer}>
        {chart === 'sparkline' && (
          <SparklineChart data={chartData} color={color} />
        )}
      </View>
    )}
  </View>
);
```
